#!/bin/bash

urls=$1

proxy=$(dig balrog-proxy.r53-2.services.mozilla.com +short | head -n 1)

for i in `cat $urls`; do
    curl --resolve aus3.mozilla.org:443:$proxy https://aus3.mozilla.org$i > /tmp/a.txt
    curl https://aus3.mozilla.org$i > /tmp/b.txt
    echo "Diffing output from URLs with path: $i"
    diff -Naur /tmp/a.txt /tmp/b.txt
done
